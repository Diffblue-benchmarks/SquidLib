001package com.badlogic.gdx.graphics.g2d;
002
003import com.badlogic.gdx.graphics.Color;
004import com.badlogic.gdx.graphics.glutils.ShaderProgram;
005import squidpony.squidgrid.gui.gdx.FloatFilter;
006import squidpony.squidgrid.gui.gdx.FloatFilters;
007import squidpony.squidmath.NumberTools;
008
009/**
010 * A drop-in substitute for {@link SpriteBatch} that filters any colors used to tint text or images using a
011 * {@link FloatFilter}. The filter may have an effect on speed in some cases, but even moderately complex filters like
012 * {@link FloatFilters.YCbCrFilter} seem to perform perfectly well, spiking at above 1000 FPS on SparseDemo with a
013 * filter that changes parameters.
014 * <br>
015 * This unfortunately has to be in a libGDX package because a field of SpriteBatch that is vital for the operation of
016 * this class was changed from having a trivial setter to a significantly more expensive one, and accessing the
017 * package-private field without using the pricey setter needs the accessing code to be in the same package.
018 * <br>
019 * Created by Tommy Ettinger on 12/8/2018.
020 */
021public class FilterBatch extends SpriteBatch {
022    public FloatFilter filter;
023
024    public FloatFilter getFilter() {
025        return filter;
026    }
027
028    public void setFilter(FloatFilter filter) {
029        this.filter = filter;
030    }
031
032    /**
033     * Constructs a new SpriteBatch with a size of 1000, one buffer, and the default shader.
034     *
035     * @see SpriteBatch#SpriteBatch(int, ShaderProgram)
036     */
037    public FilterBatch() {
038        super();
039        filter = new FloatFilters.IdentityFilter();
040    }
041
042    /**
043     * Constructs a SpriteBatch with one buffer and the default shader.
044     *
045     * @param size
046     * @see SpriteBatch#SpriteBatch(int, ShaderProgram)
047     */
048    public FilterBatch(int size) {
049        super(size);
050        filter = new FloatFilters.IdentityFilter();
051    }
052
053    /**
054     * Constructs a new SpriteBatch. Sets the projection matrix to an orthographic projection with y-axis point upwards, x-axis
055     * point to the right and the origin being in the bottom left corner of the screen. The projection will be pixel perfect with
056     * respect to the current screen resolution.
057     * <p>
058     * The defaultShader specifies the shader to use. Note that the names for uniforms for this default shader are different than
059     * the ones expected for shaders set with {@link #setShader(ShaderProgram)}. See {@link #createDefaultShader()}.
060     *
061     * @param size          The max number of sprites in a single batch. Max of 8191.
062     * @param defaultShader The default shader to use. This is not owned by the SpriteBatch and must be disposed separately.
063     */
064    public FilterBatch(int size, ShaderProgram defaultShader) {
065        super(size, defaultShader);
066        filter = new FloatFilters.IdentityFilter();
067    }
068
069    /**
070     * Constructs a new SpriteBatch with a size of 1000, one buffer, the default shader, and the given FloatFilter.
071     * @param filter        a {@link FloatFilter}, such as one from {@link FloatFilters}
072     * @see SpriteBatch#SpriteBatch(int, ShaderProgram)
073     */
074    public FilterBatch(FloatFilter filter) {
075        super();
076        this.filter = filter;
077    }
078
079    /**
080     * Constructs a SpriteBatch with one buffer of the given size, the default shader, and the given FloatFilter.
081     *
082     * @param size          The max number of sprites in a single batch. Max of 8191.
083     * @param filter        a {@link FloatFilter}, such as one from {@link FloatFilters}
084     * @see SpriteBatch#SpriteBatch(int, ShaderProgram)
085     */
086    public FilterBatch(int size, FloatFilter filter) {
087        super(size);
088        this.filter = filter;
089    }
090
091    /**
092     * Constructs a new SpriteBatch with the given FloatFilter.
093     * Sets the projection matrix to an orthographic projection with y-axis pointing upwards, x-axis pointing to the
094     * right and the origin being in the bottom left corner of the screen. The projection will be pixel perfect with
095     * respect to the current screen resolution.
096     * <p>
097     * The defaultShader specifies the shader to use. Note that the names for uniforms for this default shader are different than
098     * the ones expected for shaders set with {@link #setShader(ShaderProgram)}. See {@link #createDefaultShader()}.
099     *
100     * @param size          The max number of sprites in a single batch. Max of 8191.
101     * @param defaultShader The default shader to use. This is not owned by the SpriteBatch and must be disposed separately.
102     * @param filter        a {@link FloatFilter}, such as one from {@link FloatFilters}
103     */
104    public FilterBatch(int size, ShaderProgram defaultShader, FloatFilter filter) {
105        super(size, defaultShader);
106        this.filter = filter;
107    }
108
109    @Override
110    public void setColor(Color tint) {
111        colorPacked = filter.alter(tint);
112        Color.abgr8888ToColor(super.getColor(), colorPacked);
113    }
114
115    @Override
116    public void setColor(float r, float g, float b, float a) {
117        int intBits = ((int)(255 * a) & 0xFE) << 24 | (int)(255 * b) << 16 | (int)(255 * g) << 8 | (int)(255 * r);
118        colorPacked = (filter.alter(NumberTools.intBitsToFloat(intBits)));
119    }
120
121    public void setColor(float color) {
122        colorPacked = (filter.alter(color));
123    }
124
125    @Override
126    public Color getColor() {
127        Color.abgr8888ToColor(super.getColor(), getPackedColor());
128        return super.getColor();
129    }
130
131    @Override
132    public void setPackedColor(float packedColor) {
133        colorPacked = filter.alter(packedColor);
134    }
135
136    @Override
137    public float getPackedColor() {
138        return colorPacked;
139    }
140}
