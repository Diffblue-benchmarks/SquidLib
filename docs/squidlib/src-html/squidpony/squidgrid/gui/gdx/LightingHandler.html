016/**
017 * A convenience class that makes dealing with multiple colored light sources easier.
018 * All fields are public and documented to encourage their use alongside the API methods. The typical usage case for
019 * this class is when a game has complex lighting needs that should be consolidated into one LightingHandler per level,
020 * where a level corresponds to a {@code char[][]}. After constructing a LightingHandler with the resistances for a
021 * level, you should add all light sources with their positions, either using {@link #addLight(int, int, Radiance)} or
022 * by directly putting keys and values into {@link #lights}. Then you can calculate the visible cells once lighting is
023 * considered (which may include distant lit cells with unseen but unobstructing cells between the viewer and the light)
024 * using {@link #calculateFOV(Coord)}, which. You can update the flicker and strobe effects on all Radiance objects,
025 * which is typically done every frame, using {@link #update(Coord)} (usually with the player as the viewer), and once
026 * that update() call has been made you can call {@link #draw(SparseLayers)} to change the background colors of a
027 * SparseLayers, {@link #draw(SquidPanel)} to change the colors of a SquidPanel (typically the background layer of a
028 * SquidLayers, as from {@link SquidLayers#getBackgroundLayer()}), or {@link #draw(float[][])} to change a 2D float
029 * array that holds packed float colors (which may be used in some custom setup). To place user-interface lighting
030 * effects that don't affect the actual FOV of creatures in the game, you can use {@link #updateUI(Coord, Radiance)},
031 * which is called after {@link #update(Coord)} but before {@link #draw(float[][])}.
032 * <br>
033 * Created by Tommy Ettinger on 11/2/2018.
034 */
035public class LightingHandler implements Serializable {
036    private static final long serialVersionUID = 0L;
037
038    /**
039     * How light should spread; usually {@link Radius#CIRCLE} unless gameplay reasons need it to be SQUARE or DIAMOND.
040     */
041    public Radius radiusStrategy;
042    /**
043     * The 2D array of light-resistance values from 0.0 to 1.0 for each cell on the map, as produced by
044     * {@link squidpony.squidgrid.mapping.DungeonUtility#generateResistances(char[][])}.
045     */
046    public double[][] resistances;
047    /**
048     * What the "viewer" (as passed to {@link #calculateFOV(Coord)}) can see either nearby without light or because an
049     * area in line-of-sight has light in it. Edited by {@link #calculateFOV(Coord)} and {@link  #update(Coord)}, but
050     * not {@link #updateUI(Coord, Radiance)} (which is meant for effects that are purely user-interface).
051     */
052    public double[][] fovResult;
053    /**
054     * A 2D array of doubles that are either 0.0 if a cell has an obstruction between it and the viewer, or greater than
055     * 0.0 otherwise.
056     */
057    public double[][] losResult;
058    /**
059     * Temporary storage array used for calculations involving {@link #fovResult}; it sometimes may make sense for other
060     * code to use this as temporary storage as well.
061     */
062    public double[][] tempFOV;
063    /**
064     * A pair of 2D float arrays with different usages; {@code colorLighting[0]} is a 2D array that stores the strength
065     * of light in each cell, and {@code colorLighting[1]} is a 2D array that stores the color of light in each cell, as
066     * a packed float color. Both 2D arrays are the size of the map, as defined by {@link #resistances} initially and
067     * later available in {@link #width} and {@link #height}.
068     */
069    public float[][][] colorLighting;
070    /**
071     * Temporary storage array used for calculations involving {@link #colorLighting}; it sometimes may make sense for
072     * other code to use this as temporary storage as well.
073     */
074    public float[][][] tempColorLighting;
075    /**
076     * Width of the 2D arrays used in this, as obtained from {@link #resistances}.
077     */
078    public int width;
079    /**
080     * Height of the 2D arrays used in this, as obtained from {@link #resistances}.
081     */
082    public int height;
083    /**
084     * The packed float color to mix background cells with when a cell has lighting and is within line-of-sight, but has
085     * no background color to start with (its color is 0f as a packed float, or {@link SColor#TRANSPARENT}).
086     */
087    public float backgroundColor;
088    /**
089     * How far the viewer can see without light; defaults to 4.0 cells, and you are encouraged to change this member
090     * field if the vision range changes after construction.
091     */
092    public double viewerRange;
093    /**
094     * A mapping from positions as {@link Coord} objects to {@link Radiance} objects that describe the color, lighting
095     * radius, and changes over time of any in-game lights that should be shown on the map and change FOV. You can edit
096     * this manually or by using {@link #moveLight(int, int, int, int)}, {@link #addLight(int, int, Radiance)}, and
097     * {@link #removeLight(int, int)}.
098     */
099    public OrderedMap<Coord, Radiance> lights;

137     * Given a resistance array as produced by {@link squidpony.squidgrid.mapping.DungeonUtility#generateResistances(char[][])}
138     * or {@link squidpony.squidgrid.mapping.DungeonUtility#generateSimpleResistances(char[][])}, makes a
139     * LightingHandler that can have {@link Radiance} objects added to it in various locations.
140     * @param resistance a resistance array as produced by DungeonUtility
141     * @param backgroundColor the background color to use, as a packed float (produced by {@link Color#toFloatBits()})
142     * @param radiusStrategy the shape lights should take, typically {@link Radius#CIRCLE} for "realistic" lights or one
143     *                       of {@link Radius#DIAMOND} or {@link Radius#SQUARE} to match game rules for distance
144     * @param viewerVisionRange how far the player can see without light, in cells
145     */
146    public LightingHandler(double[][] resistance, float backgroundColor, Radius radiusStrategy, double viewerVisionRange)
147    {
148        this.radiusStrategy = radiusStrategy;
149        viewerRange = viewerVisionRange;
150        this.backgroundColor = backgroundColor;
151        resistances = resistance;
152        width = resistances.length;
153        height = resistances[0].length;
154        fovResult = new double[width][height];
155        tempFOV = new double[width][height];
156        losResult = new double[width][height];
157        colorLighting = SColor.blankColoredLighting(width, height);
158        tempColorLighting = new float[2][width][height];
159        Coord.expandPoolTo(width, height);
160        lights = new OrderedMap<>(32);
161    }

254    /**
255     * Edits {@link #colorLighting} by adding in and mixing the colors in {@link #tempColorLighting}, with the strength
256     * of light in tempColorLighting boosted by flare (which can be any finite float greater than -1f, but is usually
257     * from 0f to 1f when increasing strength).
258     * Primarily used internally, but exposed so outside code can do the same things this class can.
259     * @param flare boosts the effective strength of lighting in {@link #tempColorLighting}; usually from 0 to 1
260     */
261    public void mixColoredLighting(float flare)
262    {
263        float[][][] basis = colorLighting, other = tempColorLighting;
264        flare += 1f;
265        float b0, b1, o0, o1;
266        for (int x = 0; x < width; x++) {
267            for (int y = 0; y < height; y++) {
268                b0 = basis[0][x][y];
269                b1 = basis[1][x][y];
270                o0 = other[0][x][y];
271                o1 = other[1][x][y];
272                if (b1 == FLOAT_WHITE) {
273                    basis[1][x][y] = o1;
274                    basis[0][x][y] = Math.min(1.0f, b0 + o0 * flare);
275                } else {
276                    if (o1 != FLOAT_WHITE) {
277                        float change = (o0 - b0) * 0.5f + 0.5f;
278                        final int s = NumberTools.floatToIntBits(b1), e = NumberTools.floatToIntBits(o1),
279                                rs = (s & 0xFF), gs = (s >>> 8) & 0xFF, bs = (s >>> 16) & 0xFF, as = s & 0xFE000000,
280                                re = (e & 0xFF), ge = (e >>> 8) & 0xFF, be = (e >>> 16) & 0xFF, ae = (e >>> 25);
281                        change *= ae * 0.007874016f;
282                        basis[1][x][y] = NumberTools.intBitsToFloat(((int) (rs + change * (re - rs)) & 0xFF)
283                                | ((int) (gs + change * (ge - gs)) & 0xFF) << 8
284                                | (((int) (bs + change * (be - bs)) & 0xFF) << 16)
285                                | as);
286                        basis[0][x][y] = Math.min(1.0f, b0 + o0 * change * flare);
287                    } else {
288                        basis[0][x][y] = Math.min(1.0f, b0 + o0 * flare);
289                    }
290                }
291            }
292        }
293    }

297     * Typically called every frame, this updates the flicker and strobe effects of Radiance objects and applies those
298     * changes in lighting color and strength to the various fields of this LightingHandler. This method is usually
299     * called before each call to {@link #draw(float[][])}, but other code may be between the calls and may
300     * affect the lighting in customized ways.
301     * @param viewerPosition the position of the player or other viewer
302     */
303    public void update(Coord viewerPosition)
304    {
305        update(viewerPosition.x, viewerPosition.y);
306    }

308    /**
309     * Typically called every frame, this updates the flicker and strobe effects of Radiance objects and applies those
310     * changes in lighting color and strength to the various fields of this LightingHandler. This method is usually
311     * called before each call to {@link #draw(float[][])}, but other code may be between the calls and may
312     * affect the lighting in customized ways.
313     * @param viewerX the x-position of the player or other viewer
314     * @param viewerY the y-position of the player or other viewer
315     */
316    public void update(int viewerX, int viewerY)
317    {
318        Radiance radiance;
319        FOV.reuseFOV(resistances, fovResult, viewerX, viewerY, viewerRange, radiusStrategy);
320        FOV.reuseLOS(resistances, losResult, viewerX, viewerY);
321        SColor.eraseColoredLighting(colorLighting);
322        final int sz = lights.size();
323        Coord pos;
324        for (int i = 0; i < sz; i++) {
325            pos = lights.keyAt(i);
326            radiance = lights.getAt(i);
327            FOV.reuseFOV(resistances, tempFOV, pos.x, pos.y, radiance.currentRange());
328            SColor.colorLightingInto(tempColorLighting, tempFOV, radiance.color);
329            mixColoredLighting(radiance.flare);
330        }
331        for (int x = 0; x < width; x++) {
332            for (int y = 0; y < height; y++) {
333                if (losResult[x][y] > 0.0) {
334                    fovResult[x][y] = MathUtils.clamp(fovResult[x][y] + colorLighting[0][x][y], 0, 1);
335                }
336            }
337        }
338    }

397    /**
398     * Given a 2D array of packed float colors, fills the 2D array with different colors based on what lights are
399     * present in line of sight of the viewer and the various flicker or strobe effects that Radiance light sources can
400     * do. You should usually call {@link #update(int, int)} before each call to draw(), but you may want to make custom
401     * changes to the lighting in between those two calls (that is the only place those changes will be noticed).
402     * @param backgrounds a 2D float array, typically obtained from {@link squidpony.squidgrid.gui.gdx.SquidPanel#colors} or {@link squidpony.squidgrid.gui.gdx.SparseLayers#backgrounds}
403     */
404    public void draw(float[][] backgrounds)
405    {
406        float current;
407        for (int x = 0; x < width; x++) {
408            for (int y = 0; y < height; y++) {
409                if (losResult[x][y] > 0.0 && fovResult[x][y] > 0.0) {
410                        current = backgrounds[x][y];
411                        if(current == 0f)
412                            current = backgroundColor;
413                        backgrounds[x][y] = lerpFloatColorsBlended(current,
414                                colorLighting[1][x][y], colorLighting[0][x][y] * 0.4f);
415                }
416            }
417        }
418    }

432     * Used to calculate what cells are visible as if any flicker or strobe effects were simply constant light sources.
433     * Runs part of the calculations to draw lighting as if all radii are at their widest, but does no actual drawing.
434     * Sets {@link #fovResult} and {@link #losResult} based on the given viewer position and any lights.
435     * @param viewerX the x-position of the player or other viewer
436     * @param viewerY the y-position of the player or other viewer
437     * @return the calculated FOV 2D array, which is also stored in {@link #fovResult}
438     */

440    {
441        Radiance radiance;
442        FOV.reuseFOV(resistances, fovResult, viewerX, viewerY, viewerRange, radiusStrategy);
443        FOV.reuseLOS(resistances, losResult, viewerX, viewerY);
444        SColor.eraseColoredLighting(colorLighting);
445        final int sz = lights.size();
446        Coord pos;
447        for (int i = 0; i < sz; i++) {
448            pos = lights.keyAt(i);
449            radiance = lights.getAt(i);
450            FOV.reuseFOV(resistances, tempFOV, pos.x, pos.y, radiance.range);
451            SColor.colorLightingInto(tempColorLighting, tempFOV, radiance.color);
452            mixColoredLighting(radiance.flare);
453        }
454        for (int x = 0; x < width; x++) {
455            for (int y = 0; y < height; y++) {
456                if (losResult[x][y] > 0.0) {
457                    fovResult[x][y] = MathUtils.clamp(fovResult[x][y] + colorLighting[0][x][y], 0, 1);
458                }
459            }
460        }
461        return fovResult;
462    }
