128    /**
129     * Color difference metric; returns large numbers even for smallish differences.
130     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
131     *
132     * @param color1 an RGBA8888 color as an int
133     * @param color2 an RGBA8888 color as an int
134     * @return the difference between the given colors, as a positive int
135     */
136    public static int difference(final int color1, final int color2) {
137        int rmean = ((color1 >>> 24) + (color2 >>> 24)) >> 1;
138        int r = (color1 >>> 24) - (color2 >>> 24);
139        int g = (color1 >>> 16 & 0xFF) - (color2 >>> 16 & 0xFF) << 1;
140        int b = (color1 >>> 8 & 0xFF) - (color2 >>> 8 & 0xFF);
141        return (((512 + rmean) * r * r) >> 8) + g * g + (((767 - rmean) * b * b) >> 8);
142    }

144    /**
145     * Color difference metric; returns large numbers even for smallish differences.
146     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
147     *
148     * @param color1 an RGBA8888 color as an int
149     * @param r2     red value from 0 to 255, inclusive
150     * @param g2     green value from 0 to 255, inclusive
151     * @param b2     blue value from 0 to 255, inclusive
152     * @return the difference between the given colors, as a positive int
153     */
154    public static int difference(final int color1, int r2, int g2, int b2) {
155//        r2 = (r2 << 3 | r2 >>> 2);
156//        g2 = (g2 << 3 | g2 >>> 2);
157//        b2 = (b2 << 3 | b2 >>> 2);
158        final int rmean = ((color1 >>> 24) + r2) >> 1,
159                r = (color1 >>> 24) - r2,
160                g = (color1 >>> 16 & 0xFF) - g2 << 1,
161                b = (color1 >>> 8 & 0xFF) - b2;
162        return (((512 + rmean) * r * r) >> 8) + g * g + (((767 - rmean) * b * b) >> 8);
163    }

165    /**
166     * Color difference metric; returns large numbers even for smallish differences.
167     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
168     *
169     * @param r1 red value from 0 to 255, inclusive
170     * @param r2 red value from 0 to 255, inclusive
171     * @param g1 green value from 0 to 255, inclusive
172     * @param g2 green value from 0 to 255, inclusive
173     * @param b1 blue value from 0 to 255, inclusive
174     * @param b2 blue value from 0 to 255, inclusive
175     * @return the difference between the given colors, as a positive int
176     */
177    public static int difference(final int r1, final int r2, final int g1, final int g2, final int b1, final int b2) {
178        final int rmean = (r1 + r2) >> 1,
179                r = r1 - r2,
180                g = g1 - g2 << 1,
181                b = b1 - b2;
182        return (((512 + rmean) * r * r) >> 8) + g * g + (((767 - rmean) * b * b) >> 8);
183    }

185    /**
186     * Gets a pseudo-random float between -0.3f and 0.7f, determined by the lower 23 bits and upper 2 bits of seed.
187     * The average value this returns will be less than 0f, despite its upper bound being further from 0 than its
188     * lower bound, because values between 0.2f and 0.7f are a third as likely to appear as values less than 0.125f,
189     * and even for values less than 0.2f, the range between -0.05f and 0.2f is a third as likely as the range from
190     * -0.3f to 0.05f. This introduced bias can be useful for adding some order to otherwise-random dithering.
191     * @param seed any int, but only the least-significant 23 bits will be used
192     * @return a float between -0.3f and 0.7f, weighted toward the lower end of the range
193     */
194    public static float randomXi(int seed)
195    {
196        return NumberUtils.intBitsToFloat((seed & 0x7FFFFF & ((seed >>> 11 & 0x600000)|0x1FFFFF)) | 0x3f800000) - 1.3f;
197    }

199    /**
200     * Builds the palette information this PNG8 stores from the RGBA8888 ints in {@code rgbaPalette}, up to 256 colors.
201     * Alpha is not preserved except for the first item in rgbaPalette, and only if it is {@code 0} (fully transparent
202     * black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, or only has one color, then
203     * this defaults to DawnBringer's Aurora palette with 256 hand-chosen colors (including transparent).
204     *
205     * @param rgbaPalette an array of RGBA8888 ints; all will be used up to 256 items or the length of the array
206     */
207    public void exact(int[] rgbaPalette) {
208        if (rgbaPalette == null || rgbaPalette.length < 2) {
209            rgbaPalette = auroraPalette;
210        }
211        Arrays.fill(paletteArray, 0);
212        Arrays.fill(paletteMapping, (byte) 0);
213        final int plen = Math.min(256, rgbaPalette.length);
214        int color, c2;
215        int dist;
216        for (int i = 0; i < plen; i++) {
217            color = rgbaPalette[i];
218            paletteArray[i] = color;
219            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
220        }
221        int rr, gg, bb;
222        for (int r = 0; r < 32; r++) {
223            rr = (r << 3 | r >>> 2);
224            for (int g = 0; g < 32; g++) {
225                gg = (g << 3 | g >>> 2);
226                for (int b = 0; b < 32; b++) {
227                    c2 = r << 10 | g << 5 | b;
228                    if (paletteMapping[c2] == 0) {
229                        bb = (b << 3 | b >>> 2);
230                        dist = 0x7FFFFFFF;
231                        for (int i = 1; i < plen; i++) {
232                            if (dist > (dist = Math.min(dist, difference(paletteArray[i], rr, gg, bb))))
233                                paletteMapping[c2] = (byte) i;
234                        }
235                    }
236                }
237            }
238        }
239    }

255     * Builds the palette information this PNG8 stores from the Color objects in {@code colorPalette}, up to 256 colors.
256     * Alpha is not preserved except for the first item in colorPalette, and only if its r, g, b, and a values are all
257     * 0f (fully transparent black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, only has
258     * one color, or limit is less than 2, then this defaults to DawnBringer's Aurora palette with 256 hand-chosen
259     * colors (including transparent).
260     *
261     * @param colorPalette an array of Color objects; all will be used up to 256 items, limit, or the length of the array
262     * @param limit        a limit on how many Color items to use from colorPalette; useful if colorPalette is from an Array
263     */
264    public void exact(Color[] colorPalette, int limit) {
265        if (colorPalette == null || colorPalette.length < 2 || limit < 2) {
266            exact(auroraPalette);
267            return;
268        }
269        Arrays.fill(paletteArray, 0);
270        Arrays.fill(paletteMapping, (byte) 0);
271        final int plen = Math.min(Math.min(256, colorPalette.length), limit);
272        int color, c2;
273        int dist;
274        for (int i = 0; i < plen; i++) {
275            color = Color.rgba8888(colorPalette[i]);
276            paletteArray[i] = color;
277            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
278        }
279        int rr, gg, bb;
280        for (int r = 0; r < 32; r++) {
281            rr = (r << 3 | r >>> 2);
282            for (int g = 0; g < 32; g++) {
283                gg = (g << 3 | g >>> 2);
284                for (int b = 0; b < 32; b++) {
285                    c2 = r << 10 | g << 5 | b;
286                    if (paletteMapping[c2] == 0) {
287                        bb = (b << 3 | b >>> 2);
288                        dist = 0x7FFFFFFF;
289                        for (int i = 1; i < plen; i++) {
290                            if (dist > (dist = Math.min(dist, difference(paletteArray[i], rr, gg, bb))))
291                                paletteMapping[c2] = (byte) i;
292                        }
293                    }
294                }
295            }
296        }

298    /**
299     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
300     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
301     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
302     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
303     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
304     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
305     * uses a threshold value to determine whether it should permit a less-common color into the palette, and if the
306     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least 400, it is
307     * allowed in the palette, otherwise it is kept out for being too similar to existing colors. This doesn't return a
308     * value but instead stores the palette info in this object; a PaletteReducer can be assigned to the
309     * {@link PNG8#palette} field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
310     *
311     * @param pixmap a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
312     */
313    public void analyze(Pixmap pixmap) {
314        analyze(pixmap, 400);
315    }

319        public int compare(IntIntMap.Entry o1, IntIntMap.Entry o2) {
320            return o2.value - o1.value;
321        }

325    /**
326     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
327     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
328     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
329     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
330     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
331     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
332     * takes a threshold value to determine whether it should permit a less-common color into the palette, and if the
333     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least
334     * {@code threshold}, it is allowed in the palette, otherwise it is kept out for being too similar to existing
335     * colors. The threshold is usually between 250 and 1000, and 400 is a good default. This doesn't return a value but
336     * instead stores the palette info in this object; a PaletteReducer can be assigned to the {@link PNG8#palette}
337     * field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
338     *
339     * @param pixmap    a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
340     * @param threshold a minimum color difference as produced by {@link #difference(int, int)}; usually between 250 and 1000, 400 is a good default
341     */
342    public void analyze(Pixmap pixmap, int threshold) {
343        Arrays.fill(paletteArray, 0);
344        Arrays.fill(paletteMapping, (byte) 0);
345        int color;
346        final int width = pixmap.getWidth(), height = pixmap.getHeight();
347        IntIntMap counts = new IntIntMap(256);
348        int hasTransparent = 0;
349        int[] reds = new int[256], greens = new int[256], blues = new int[256];
350        for (int y = 0; y < height; y++) {
351            for (int x = 0; x < width; x++) {
352                color = pixmap.getPixel(x, y);
353                if ((color & 0x80) != 0) {
354                    color |= (color >>> 5 & 0x07070700) | 0xFE;
355                    counts.getAndIncrement(color, 0, 1);
356                } else {
357                    hasTransparent = 1;
358                }
359            }
360        }
361        final int cs = counts.size;
362        if (cs + hasTransparent <= 256) {
363            int i = hasTransparent;
364            for(IntIntMap.Entry e : counts) {
365                color = e.key;
366                paletteArray[i] = color;
367                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
368                paletteMapping[color] = (byte) i;
369                reds[i] = color >>> 10;
370                greens[i] = color >>> 5 & 31;
371                blues[i] = color & 31;
372                i++;
373            }
374        } else // reduce color count
375        {
376            ArrayList<IntIntMap.Entry> es = new ArrayList<>(cs);
377            for(IntIntMap.Entry e : counts)
378            {
379                IntIntMap.Entry e2 = new IntIntMap.Entry();
380                e2.key = e.key;
381                e2.value = e.value;
382                es.add(e2);
383            }
384            Collections.sort(es, entryComparator);
385            int i = 1, c = 0;
386            PER_BEST:
387            for (; i < 256 && c < cs;) {
388                color = es.get(c++).key;
389                for (int j = 1; j < i; j++) {
390                    if (difference(color, paletteArray[j]) < threshold)
391                        continue PER_BEST;
392                }
393                paletteArray[i] = color;
394                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
395                paletteMapping[color] = (byte) i;
396                reds[i] = color >>> 10;
397                greens[i] = color >>> 5 & 31;
398                blues[i] = color & 31;
399                i++;
400            }
401        }
402        int c2, dist;
403        for (int r = 0; r < 32; r++) {
404            for (int g = 0; g < 32; g++) {
405                for (int b = 0; b < 32; b++) {
406                    c2 = r << 10 | g << 5 | b;
407                    if (paletteMapping[c2] == 0) {
408                        dist = 0x7FFFFFFF;
409                        for (int i = 1; i < 256; i++) {
410                            if (dist > (dist = Math.min(dist, difference(reds[i], r, greens[i], g, blues[i], b))))
411                                paletteMapping[c2] = (byte) i;
412                        }
413                    }
414                }
415            }
416        }
417    }

419    /**
420     * Changes the "strength" of the dither effect applied during {@link #reduce(Pixmap)} calls. The default is 1f,
421     * and while both values higher than 1f and lower than 1f are valid, they should not be negative. If you want dither
422     * to be eliminated, don't set dither strength to 0; use {@link #reduceSolid(Pixmap)} instead of reduce().
423     * @param ditherStrength dither strength as a non-negative float that should be close to 1f
424     */

426        this.ditherStrength = 0.5f * ditherStrength;
427        this.halfDitherStrength = 0.25f * ditherStrength;

430    /**
431     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, dithering when it can.
432     * If you want to reduce the colors in a Pixmap based on what it currently contains, call
433     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
434     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;
435     * {@link #exact(int[])} is the tool for that job.
436     * <br>
437     * This method is not incredibly fast because of the extra calculations it has to do for dithering, but if you can
438     * compute the PaletteReducer once and reuse it, that will save some time.
439     * @param pixmap a Pixmap that will be modified in place
440     * @return the given Pixmap, for chaining
441     */
442    public Pixmap reduce (Pixmap pixmap) {
443        boolean hasTransparent = (paletteArray[0] == 0);
444        final int lineLen = pixmap.getWidth(), h = pixmap.getHeight();
445        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
446        if (curErrorRedBytes == null) {
447            curErrorRed = (curErrorRedBytes = new ByteArray(lineLen)).items;
448            nextErrorRed = (nextErrorRedBytes = new ByteArray(lineLen)).items;
449            curErrorGreen = (curErrorGreenBytes = new ByteArray(lineLen)).items;
450            nextErrorGreen = (nextErrorGreenBytes = new ByteArray(lineLen)).items;
451            curErrorBlue = (curErrorBlueBytes = new ByteArray(lineLen)).items;
452            nextErrorBlue = (nextErrorBlueBytes = new ByteArray(lineLen)).items;
453        } else {
454            curErrorRed = curErrorRedBytes.ensureCapacity(lineLen);
455            nextErrorRed = nextErrorRedBytes.ensureCapacity(lineLen);
456            curErrorGreen = curErrorGreenBytes.ensureCapacity(lineLen);
457            nextErrorGreen = nextErrorGreenBytes.ensureCapacity(lineLen);
458            curErrorBlue = curErrorBlueBytes.ensureCapacity(lineLen);
459            nextErrorBlue = nextErrorBlueBytes.ensureCapacity(lineLen);
460            for (int i = 0; i < lineLen; i++) {
461                nextErrorRed[i] = 0;
462                nextErrorGreen[i] = 0;
463                nextErrorBlue[i] = 0;
464            }
465
466        }
467        Pixmap.Blending blending = pixmap.getBlending();
468        pixmap.setBlending(Pixmap.Blending.None);
469        int color, used, rdiff, gdiff, bdiff;
470        byte er, eg, eb, paletteIndex;
471        for (int y = 0; y < h; y++) {
472            int ny = y + 1;
473            for (int i = 0; i < lineLen; i++) {
474                curErrorRed[i] = nextErrorRed[i];
475                curErrorGreen[i] = nextErrorGreen[i];
476                curErrorBlue[i] = nextErrorBlue[i];
477                nextErrorRed[i] = 0;
478                nextErrorGreen[i] = 0;
479                nextErrorBlue[i] = 0;
480            }
481            for (int px = 0; px < lineLen; px++) {
482                color = pixmap.getPixel(px, y) & 0xF8F8F880;
483                if ((color & 0x80) == 0 && hasTransparent)
484                    pixmap.drawPixel(px, y, 0);
485                else {
486                    er = curErrorRed[px];
487                    eg = curErrorGreen[px];
488                    eb = curErrorBlue[px];
489                    color |= (color >>> 5 & 0x07070700) | 0xFE;
490                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
491                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
492                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
493                    paletteIndex =
494                            paletteMapping[((rr << 7) & 0x7C00)
495                                    | ((gg << 2) & 0x3E0)
496                                    | ((bb >>> 3))];
497                    used = paletteArray[paletteIndex & 0xFF];
498                    pixmap.drawPixel(px, y, used);
499                    rdiff = (color>>>24)-    (used>>>24);
500                    gdiff = (color>>>16&255)-(used>>>16&255);
501                    bdiff = (color>>>8&255)- (used>>>8&255);
502                    if(px < lineLen - 1)
503                    {
504                        curErrorRed[px+1]   += rdiff * ditherStrength;
505                        curErrorGreen[px+1] += gdiff * ditherStrength;
506                        curErrorBlue[px+1]  += bdiff * ditherStrength;
507                    }
508                    if(ny < h)
509                    {
510                        if(px > 0)
511                        {
512                            nextErrorRed[px-1]   += rdiff * halfDitherStrength;
513                            nextErrorGreen[px-1] += gdiff * halfDitherStrength;
514                            nextErrorBlue[px-1]  += bdiff * halfDitherStrength;
515                        }
516                        nextErrorRed[px]   += rdiff * halfDitherStrength;
517                        nextErrorGreen[px] += gdiff * halfDitherStrength;
518                        nextErrorBlue[px]  += bdiff * halfDitherStrength;
519                    }
520                }
521            }
522
523        }
524        pixmap.setBlending(blending);
525        return pixmap;
526    }

703     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, dithering when it can using a
704     * modified version of the algorithm presented in "Simple gradient-based error-diffusion method" by Xaingyu Y. Hu in
705     * the Journal of Electronic Imaging, 2016. This algorithm uses pseudo-randomly-generated noise to adjust
706     * Floyd-Steinberg dithering, with input for the pseudo-random state obtained by the non-transparent color values as
707     * they are encountered. Very oddly, this tends to produce less random-seeming dither than
708     * {@link #reduceBurkes(Pixmap)}, with this method often returning regular checkerboards where Burkes may produce
709     * splotches of color. If you want to reduce the colors in a Pixmap based on what it currently contains, call
710     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
711     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;
712     * {@link #exact(int[])} is the tool for that job.
713     * <br>
714     * This method is not incredibly fast because of the extra calculations it has to do for dithering, but if you can
715     * compute the PaletteReducer once and reuse it, that will save some time. This method is probably slower than
716     * {@link #reduceBurkes(Pixmap)} even though Burkes propagates error to more pixels, because this method also has to
717     * generate two random values per non-transparent pixel. The random number "algorithm" this uses isn't very good
718     * because it doesn't have to be good, it should just be fast and avoid clear artifacts; it's similar to one of
719     * <a href="http://www.drdobbs.com/tools/fast-high-quality-parallel-random-number/231000484?pgno=2">Mark Overton's
720     * subcycle generators</a> (which are usually paired, but that isn't the case here), but because it's
721     * constantly being adjusted by additional colors as input, it may be more comparable to a rolling hash. This uses
722     * {@link #randomXi(int)} to get the parameter in Hu's paper that's marked as {@code aξ}, but our randomXi() is
723     * adjusted so it has a higher range into positive values but produces them less frequently than negative ones. That
724     * quirk ends up getting rather high quality for this method.
725     * @param pixmap a Pixmap that will be modified in place
726     * @return the given Pixmap, for chaining
727     */
728    public Pixmap reduceWithNoise (Pixmap pixmap) {
729        boolean hasTransparent = (paletteArray[0] == 0);
730        final int lineLen = pixmap.getWidth(), h = pixmap.getHeight();
731        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
732        if (curErrorRedBytes == null) {
733            curErrorRed = (curErrorRedBytes = new ByteArray(lineLen)).items;
734            nextErrorRed = (nextErrorRedBytes = new ByteArray(lineLen)).items;
735            curErrorGreen = (curErrorGreenBytes = new ByteArray(lineLen)).items;
736            nextErrorGreen = (nextErrorGreenBytes = new ByteArray(lineLen)).items;
737            curErrorBlue = (curErrorBlueBytes = new ByteArray(lineLen)).items;
738            nextErrorBlue = (nextErrorBlueBytes = new ByteArray(lineLen)).items;
739        } else {
740            curErrorRed = curErrorRedBytes.ensureCapacity(lineLen);
741            nextErrorRed = nextErrorRedBytes.ensureCapacity(lineLen);
742            curErrorGreen = curErrorGreenBytes.ensureCapacity(lineLen);
743            nextErrorGreen = nextErrorGreenBytes.ensureCapacity(lineLen);
744            curErrorBlue = curErrorBlueBytes.ensureCapacity(lineLen);
745            nextErrorBlue = nextErrorBlueBytes.ensureCapacity(lineLen);
746            for (int i = 0; i < lineLen; i++) {
747                nextErrorRed[i] = 0;
748                nextErrorGreen[i] = 0;
749                nextErrorBlue[i] = 0;
750            }
751
752        }
753        Pixmap.Blending blending = pixmap.getBlending();
754        pixmap.setBlending(Pixmap.Blending.None);
755        int color, used, rdiff, gdiff, bdiff, state = 0xFEEDBEEF;
756        byte er, eg, eb, paletteIndex;
757        //float xir1, xir2, xig1, xig2, xib1, xib2, // would be used if random factors were per-channel
758                // used now, where random factors are determined by whole colors as ints
759        float xi1, xi2, w1 = ditherStrength * 0.125f, w3 = w1 * 3f, w5 = w1 * 5f, w7 = w1 * 7f;
760        for (int y = 0; y < h; y++) {
761            int ny = y + 1;
762            for (int i = 0; i < lineLen; i++) {
763                curErrorRed[i] = nextErrorRed[i];
764                curErrorGreen[i] = nextErrorGreen[i];
765                curErrorBlue[i] = nextErrorBlue[i];
766                nextErrorRed[i] = 0;
767                nextErrorGreen[i] = 0;
768                nextErrorBlue[i] = 0;
769            }
770            for (int px = 0; px < lineLen; px++) {
771                color = pixmap.getPixel(px, y) & 0xF8F8F880;
772                if ((color & 0x80) == 0 && hasTransparent)
773                    pixmap.drawPixel(px, y, 0);
774                else {
775                    er = curErrorRed[px];
776                    eg = curErrorGreen[px];
777                    eb = curErrorBlue[px];
778                    color |= (color >>> 5 & 0x07070700) | 0xFE;
779                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
780                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
781                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
782                    paletteIndex =
783                            paletteMapping[((rr << 7) & 0x7C00)
784                                    | ((gg << 2) & 0x3E0)
785                                    | ((bb >>> 3))];
786                    used = paletteArray[paletteIndex & 0xFF];
787                    pixmap.drawPixel(px, y, used);
788                    rdiff = (color>>>24)-    (used>>>24);
789                    gdiff = (color>>>16&255)-(used>>>16&255);
790                    bdiff = (color>>>8&255)- (used>>>8&255);
791                    state += (color + 0x41C64E6D) ^ color >>> 7;
792                    state = (state << 21 | state >>> 11);
793                    xi1 = randomXi(state);
794                    state = (state << 15 | state >>> 17) ^ 0x9E3779B9;
795                    xi2 = randomXi(state);
796
797//                    state += rdiff ^ rdiff << 9;
798//                    state = (state << 21 | state >>> 11);
799//                    xir1 = randomXi(state);
800//                    state = (state << 21 | state >>> 11);
801//                    xir2 = randomXi(state);
802//                    state += gdiff ^ gdiff << 9;
803//                    state = (state << 21 | state >>> 11);
804//                    xig1 = randomXi(state);
805//                    state = (state << 21 | state >>> 11);

809//                    xib1 = randomXi(state);
810//                    state = (state << 21 | state >>> 11);
811//                    xib2 = randomXi(state);
812                    if(px < lineLen - 1)
813                    {
814                        curErrorRed[px+1]   += rdiff * w7 * (1f + xi1);
815                        curErrorGreen[px+1] += gdiff * w7 * (1f + xi1);
816                        curErrorBlue[px+1]  += bdiff * w7 * (1f + xi1);
817                    }
818                    if(ny < h)
819                    {
820                        if(px > 0)
821                        {
822                            nextErrorRed[px-1]   += rdiff * w3 * (1f + xi2);
823                            nextErrorGreen[px-1] += gdiff * w3 * (1f + xi2);
824                            nextErrorBlue[px-1]  += bdiff * w3 * (1f + xi2);
825                        }
826                        if(px < lineLen - 1)
827                        {
828                            nextErrorRed[px+1]   += rdiff * w1 * (1f - xi2);
829                            nextErrorGreen[px+1] += gdiff * w1 * (1f - xi2);
830                            nextErrorBlue[px+1]  += bdiff * w1 * (1f - xi2);
831                        }
832                        nextErrorRed[px]   += rdiff * w5 * (1f - xi1);
833                        nextErrorGreen[px] += gdiff * w5 * (1f - xi1);
834                        nextErrorBlue[px]  += bdiff * w5 * (1f - xi1);
835                    }
836                }
837            }
838
839        }
840        pixmap.setBlending(blending);
841        return pixmap;
842    }
