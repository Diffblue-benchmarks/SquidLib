001package squidpony.squidmath;
002
003import java.io.Serializable;
004import java.util.Arrays;
005import java.util.Random;
006
007/**
008 * A mid-quality but very fast RNG that has no apparent visual artifacts here; uses Mark Overton's CMRES subcycle
009 * generator type, with minor modifications. This is meant to be an answer to when people ask for a bare-minimum
010 * generator that's still "good enough" for games. It still passes at least 512GB of PractRand testing, which is good
011 * for a generator this small and simple. It has an unknown period that is fairly high; unless the seed used
012 * puts the generator in a worse cycle (some of which  have a much lower period, like the seed 0), the period probably
013 * won't be exhausted without hours (possibly days) of pure random number generation. It cannot produce all possible
014 * longs in its longest cycle, and it can't produce even a fraction of all possible longs in its smallest cycle. It
015 * implements RandomnessSource, but if you just want to copy this class with no dependencies, then the class declaration
016 * can easily be changed to {@code public class BasicRandom64 extends Random implements Serializable} without any other
017 * changes. Note, it does extend java.util.Random for additional ease of integration, but doesn't use the slow
018 * {@code synchronized} keyword that Random's implementations do.
019 * <br>
020 * <a href="http://www.drdobbs.com/tools/fast-high-quality-parallel-random-number/231000484">This Dr. Dobb's article has
021 * more on this type of generator</a>. The code associated with it seems to be mostly accurate, but the period on this
022 * 64-bit CMRES generator seems to be understated in the code documentation; my tests seem to show that the period is
023 * either much longer, or that due to the irreversible nature of the generator, it can't return to its original state.
024 * @author Mark Overton
025 * @author Tommy Ettinger
026 */
027public class BasicRandom64 extends Random implements RandomnessSource, Serializable {
028    private static final long serialVersionUID = 1L;
029    public long state;
030
031    public BasicRandom64()
032    {
033        state = 1L;
034    }
035
036    public BasicRandom64(final long seed) {
037        setState(seed);
038    }
039
040    public void setState(final long seed)
041    {
042        state = (seed ^ 0x9E3779B97F4A7C15L) * 0x41C64E6B ^ (seed >>> 32);
043    }
044
045    public final long nextLong() {
046        final long x = state * 0x7CDD0CFB6AD2A499L;
047        return (state -= (x << 35 | x >> 29)) + x;
048    }
049
050    /**
051     * Gets an int with at most the specified amount of bits; don't confuse this with {@link #nextInt(int)}, which gets
052     * a number between 0 and its int argument, where this draws from a different (larger) range of random results. For
053     * example, {@code next(2)} can return any 2-bit int,
054     * which is limited to 0, 1, 2, or 3. Note that if you request 0 bits, this can give you any int (32 bits).
055     * @param bits the number of bits to get, from 1 to 32
056     * @return an int with at most the specified bits
057     */
058    public final int next(final int bits) {
059        final long x = state * 0x7CDD0CFB6AD2A499L;
060        return (int)((state -= (x << 35 | x >> 29)) + x >>> (64 - bits));
061    }
062    public final int nextInt() {
063        final long x = state * 0x7CDD0CFB6AD2A499L;
064        return (int)((state -= (x << 35 | x >> 29)) + x);
065    }
066
067    /**
068     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
069     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
070     * <br>
071     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
072     *
073     * @param bound the outer bound (exclusive), can be negative or positive
074     * @return the found number
075     */
076    public int nextInt(final int bound) {
077        return (int) ((bound * (nextLong() & 0xFFFFFFFFL)) >> 32);
078    }
079    /**
080     * Exclusive on bound (which must be positive), with an inner bound of 0.
081     * If bound is negative or 0 this always returns 0.
082     * <br>
083     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
084     * with some adaptation for signed long values and a 64-bit generator. It also always gets
085     * exactly two random numbers, so it advances the state as much as {@link #nextInt(int)}.
086     * @param bound the outer exclusive bound; should be positive, otherwise this always returns 0L
087     * @return a random long between 0 (inclusive) and bound (exclusive)
088     */
089    public long nextLong(long bound) {
090        long rand = nextLong();
091        if (bound <= 0) return 0;
092        final long boundLow = bound & 0xFFFFFFFFL;
093        final long randLow = rand & 0xFFFFFFFFL;
094        final long z = (randLow * boundLow >> 32);
095        bound >>= 32;
096        rand >>>= 32;
097        final long t = rand * boundLow + z;
098        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
099    }
100
101    /**
102     * Sets the seed using a long, passing its argument to {@link #setState(long)}. That method just sets the public
103     * field {@link #state} to its argument currently, but it may do more to ensure cycle length in the future.
104     * @param seed the initial seed
105     */
106    @Override
107    public void setSeed(long seed) {
108        setState(seed);
109    }
110    /**
111     * Mutates the array arr by switching the contents at pos1 and pos2.
112     * @param arr an array of T; must not be null
113     * @param pos1 an index into arr; must be at least 0 and no greater than arr.length
114     * @param pos2 an index into arr; must be at least 0 and no greater than arr.length
115     */
116    protected static <T> void swap(T[] arr, int pos1, int pos2) {
117        final T tmp = arr[pos1];
118        arr[pos1] = arr[pos2];
119        arr[pos2] = tmp;
120    }
121
122    /**
123     * Shuffle an array using the Fisher-Yates algorithm and returns a shuffled copy, freshly-allocated, without
124     * modifying elements.
125     * <br>
126     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
127     *
128     * @param elements an array of T; will not be modified
129     * @return a shuffled copy of elements
130     */
131    public <T> T[] shuffle(T[] elements) {
132        final int size = elements.length;
133        final T[] array = Arrays.copyOf(elements, size);
134        for (int i = size; i > 1; i--) {
135            swap(array, i - 1, nextInt(i));
136        }
137        return array;
138    }
139
140    /**
141     * Shuffles an array in-place using the Fisher-Yates algorithm, affecting indices from 0 (inclusive) to length
142     * (exclusive). May be useful with libGDX Array instances, which can be shuffled with
143     * {@code random.shuffleInPlace(arr.items, arr.size)}. If you don't want the array modified, use
144     * {@link #shuffle(Object[])}.
145     * <br>
146     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
147     *
148     * @param elements an array of T; <b>will</b> be modified
149     * @return elements after shuffling it in-place
150     */
151    public <T> T[] shuffleInPlace(T[] elements, int length) {
152        final int size = Math.min(elements.length, length);
153        for (int i = size; i > 1; i--) {
154            swap(elements, i - 1, nextInt(i));
155        }
156        return elements;
157    }
158
159    /**
160     * Shuffles an array in-place using the Fisher-Yates algorithm.
161     * If you don't want the array modified, use {@link #shuffle(Object[])}.
162     * <br>
163     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
164     *
165     * @param elements an array of T; <b>will</b> be modified
166     * @return elements after shuffling it in-place
167     */
168    public <T> T[] shuffleInPlace(T[] elements) {
169        final int size = elements.length;
170        for (int i = size; i > 1; i--) {
171            swap(elements, i - 1, nextInt(i));
172        }
173        return elements;
174    }
175
176    public BasicRandom64 copy() {
177        BasicRandom64 sr = new BasicRandom64();
178        sr.state = state;
179        return sr;
180    }
181}
