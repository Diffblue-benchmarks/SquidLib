005/**
006 * The fastest generator in this library on desktop JVMs; one of Mark Overton's subcycle generators from
007 * <a href="http://www.drdobbs.com/tools/229625477">this article</a>, specifically a CMR with a 64-bit state, that has
008 * its result multiplied by a constant. Its period is unknown, but is at the very least 2 to the 42, since the generator
009 * passes PractRand after generating that many 64-bit integers (it passes with two minor anomalies, and none at the end,
010 * the 32TB mark). It probably won't pass many tests when the bits are reversed, so that is something to be aware of.
011 * <br>
012 * Notably, this generator's {@link #nextLong()} method is extremely small (as are all of the methods that use it as a
013 * basis), which may help with inlining decisions for HotSpot. Generating the next step just needs a bitwise rotation of
014 * the current state, multiplying the result by a 32-bit constant, and assigning that to state. Generating a long after
015 * that only needs a multiplication by a 31-bit constant, which could be modified to allow this to pass tests when the
016 * bits are reversed (presumably by appending one or more xorshift operations). The choice of constants for the
017 * multipliers and for the rotation needs to be carefully verified; earlier choices came close to failing PractRand at
018 * 8TB (and were worsening, so were likely to fail at 16TB), but this set of constants has higher quality in testing.
019 * For transparency, the constants used are the state multiplier 0x9E3779B9L, which is 2 to the 32 divided by the golden
020 * ratio, the post-processing multiplier 0x41C64E6DL, which was recommended in PractRand as a small LCG multiplier, and
021 * a left rotation constant of 21, which was chosen because it is slightly smaller than 1/3 of 64, and that seems to
022 * work well in a 64-bit CMR generator.
023 * <br>
024 * This is a RandomnessSource but not a StatefulRandomness because it needs to take care and avoid seeds that would put
025 * it in a short-period subcycle. It skips at most 16712703 times into its core generator's cycle when seeding. It uses
026 * constants to store 256 known midpoints for the generator, then skips an additional up-to-1023 times past that point.
027 * Each midpoint is 65536 generations ahead of the previous midpoint. There are 2 to the 18 possible starting states for
028 * this generator when using {@link #seed(int)}, but it is unknown if that method actually puts the generator in the
029 * longest possible cycle, or just a sufficiently long one.
030 * <br>
031 * The name comes from M. Overton, who discovered this category of subcycle generators, and also how this generator can
032 * really move when it comes to speed. This generator has less state than {@link Mover64RNG}, has a shorter period than
033 * it, and is faster than it in all aspects except the time needed to {@link #seed(int)}.
034 * <br>
035 * Created by Tommy Ettinger on 11/26/2018.
036 * @author Mark Overton
037 * @author Tommy Ettinger
038 */
039public final class MiniMover64RNG implements RandomnessSource {
040    private long state;
041    public MiniMover64RNG()
042    {
043        seed((int)((Math.random() * 2.0 - 1.0) * 0x80000000));
044    }
045    public MiniMover64RNG(final int state)
046    {
047        seed(state);
048    }
049
050    /**
051     * Not advised for external use; prefer {@link #MiniMover64RNG(int)} because it guarantees a good subcycle. This
052     * constructor allows all subcycles to be produced, including ones with a shorter period.
053     * @param state the state to use, exactly unless it is 0 (then this uses 1)
054     */
055    public MiniMover64RNG(final long state)
056    {
057        this.state = state == 0L ? 1L : state;
058    }

095    /**
096     * Seeds the state using 18 bits of the given int {@code s}. Although 262144 seeds are possible, this will only
097     * generate a new state at most 1023 times, and each generated state takes less time than {@link #nextLong()}.
098     * Giving this method sequential values for s will be guaranteed to produce larger distances between the sequences
099     * produced by those states; the worst-case is when s values are separated by exactly 256. For most values of s and
100     * s + 1, the sequence starting with s + 1 is the same as the sequence starting with s after it generates 65536 long
101     * values. However, the sequence starting with s + 256 is the same as the sequence starting with s after it
102     * generates 1 long value.
103     * @param s only 18 bits are used (values 0 to 262143 inclusive will all have different results).
104     */
105    public final void seed(final int s) {
106        long v = starting[s & 0xFF];
107        for (int i = s >>> 8 & 0x3FF; i > 0; i--) {
108            v = (v << 21 | v >>> 43) * 0x9E3779B9L;
109        }
110        state = v;
111    }
112
113    public final int nextInt()
114    {
115        return (int)((state = (state << 21 | state >>> 43) * 0x9E3779B9L) * 0x41C64E6DL);
116    }
117    @Override
118    public final int next(final int bits)
119    {
120        return (int)((state = (state << 21 | state >>> 43) * 0x9E3779B9L) * 0x41C64E6DL) >>> (32 - bits);
121    }
122    @Override
123    public final long nextLong()
124    {
125        return (state = (state << 21 | state >>> 43) * 0x9E3779B9L) * 0x41C64E6DL;
126    }

128    /**
129     * Produces a copy of this MiniMover64RNG that, if next() and/or nextLong() are called on this object and the
130     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
131     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
132     *
133     * @return a copy of this MiniMover64RNG
134     */
135    @Override
136    public MiniMover64RNG copy() {
137        return new MiniMover64RNG(state);
138    }

141     * Gets the state; if this generator was set with {@link #MiniMover64RNG()},
142     * {@link #MiniMover64RNG(int)}, or {@link #seed(int)}, then this will be on the optimal subcycle, otherwise it

144     * @return the state, a long
145     */

148        return state;
149    }

151    /**
152     * Sets the state to any long, which may put the generator in a low-period subcycle.
153     * Use {@link #seed(int)} to guarantee a good subcycle.
154     * @param state any int
155     */
156    public void setState(final long state)
157    {
158        this.state = state == 0L ? 1L : state;
159    }

163        return "MiniMover64RNG with state 0x" + StringKit.hex(state);
164    }

166    @Override
167    public boolean equals(Object o) {
168        if (this == o) return true;
169        if (o == null || getClass() != o.getClass()) return false;
170
171        MiniMover64RNG miniMover64RNG = (MiniMover64RNG) o;
172
173        return state == miniMover64RNG.state;
174    }

176    @Override
177    public int hashCode() {
178        return (int)(state ^ state >>> 32);
179    }
