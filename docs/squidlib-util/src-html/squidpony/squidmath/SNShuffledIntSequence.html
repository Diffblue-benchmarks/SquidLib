001package squidpony.squidmath;
002
003import java.io.Serializable;
004
005/**
006 * An infinite sequence of pseudo-random ints (typically used as indices) from 0 to some bound, with all possible ints
007 * returned in a shuffled order before re-shuffling for the next result. Does not store the sequence in memory. Uses a
008 * Swap-Or-Not network with 7 rounds using on a non-power-of-two domain, as described in
009 * <a href="https://arxiv.org/abs/1208.1176">this paper by Viet Tung Hoang, Ben Morris, and Phillip Rogaway</a>.
010 * The API is very simple; you construct a SNShuffledIntSequence by specifying how many items it should shuffle (the
011 * actual sequence is boundless, but the items it can return are limited to between 0 and some bound), and you can
012 * optionally use a seed (it will be random if you don't specify one). Call {@link #next()} on a SNShuffledIntSequence
013 * to get the next distinct int in the shuffled ordering; next() will re-shuffle the sequence if it runs out of distinct
014 * possible results. You can go back to the previous item with {@link #previous()}, which is allowed to go earlier than
015 * the first result generated (it will jump back to what is effectively the previous shuffled sequence). You can restart
016 * the sequence with {@link #restart()} to use the same sequence over again (which doesn't make much sense here, since
017 * this makes many sequences by re-shuffling), or {@link #restart(long)} to use a different seed (the bound is fixed).
018 * <br>
019 * Created by Tommy Ettinger on 10/6/2018.
020 * @author Viet Tung Hoang, Ben Morris, and Phillip Rogaway
021 * @author Tommy Ettinger
022 */
023public class SNShuffledIntSequence extends SwapOrNotShuffler implements Serializable {
024    private static final long serialVersionUID = 1L;
025    protected long seed;
026    /**
027     * Constructs a ShuffledIntSequence with a random seed and a bound of 10.
028     */
029    public SNShuffledIntSequence(){
030        this(10);
031    }
032
033    /**
034     * Constructs a ShuffledIntSequence with the given exclusive upper bound and a random seed.
035     * @param bound how many distinct ints this can return
036     */
037    public SNShuffledIntSequence(int bound)
038    {
039        this(bound, (long) ((Math.random() - 0.5) * 0x10000000000000L)
040                ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L));
041    }
042
043    /**
044     * Constructs a ShuffledIntSequence with the given exclusive upper bound and int seed.
045     * @param bound how many distinct ints this can return
046     * @param seed any int; will be used to get several seeds used internally
047     */
048    public SNShuffledIntSequence(int bound, long seed)
049    {
050        super(bound, seed);
051        this.seed = seed;
052    }
053
054    /**
055     * Gets the next distinct int in the sequence, shuffling the sequence if it has been exhausted so it never runs out.
056     * @return the next item in the sequence
057     */
058    @Override
059    public int next()
060    {
061        int shuffleIndex = super.next();
062        if(shuffleIndex == -1)
063        {
064            restart(seed += 0x9E3779B97F4A7C15L);
065            index = 0;
066            shuffleIndex = super.next();
067        }
068        return shuffleIndex;
069    }
070    /**
071     * Gets the previous returned int from the sequence (as yielded by {@link #next()}), restarting the sequence in a
072     * correctly-ordered way if it would go to before the "start" of the sequence (it is actually close to infinite both
073     * going forwards and backwards).
074     * @return the previously-given item in the sequence, or -1 if something goes wrong (which shouldn't be possible)
075     */
076    @Override
077    public int previous()
078    {
079        int shuffleIndex = super.previous();
080        if(shuffleIndex == -1)
081        {
082            restart(seed -= 0x9E3779B97F4A7C15L);
083            index = bound;
084            shuffleIndex = super.previous();
085        }
086        return shuffleIndex;
087    }
088    
089    /**
090     * Starts the sequence over, but can change the seed (completely changing the sequence). If {@code seed} is the same
091     * as the seed given in the constructor, this will use the same sequence, acting like {@link #restart()}.
092     * @param seed any int; will be used to get several seeds used internally
093     */
094    @Override
095    public void restart(long seed)
096    {
097        super.restart(seed);
098        this.seed = seed;
099    }
100}
