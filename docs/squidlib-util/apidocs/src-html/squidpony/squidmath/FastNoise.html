037/**
038 * A wide range of noise functions that can all be called from one configurable object. Originally from Jordan Peck's
039 * FastNoise library, hence the name (these functions are sometimes, but not always, very fast). This implements
040 * Noise2D, Noise3D, and Noise4D, and while 2D and 3D noise are very fast, the 4D noise is unusually slow, and normally
041 * you should use WhirlingNoise for 4D noise (for whatever reason, it's the fastest despite also taking steps to ensure
042 * higher quality). Though it doesn't implement an interface for them, you can also use this to get ridged-multi simplex

045 * and gain.

101    /**
102     * A constructor that takes a lot of parameters to specify the FastNoise from the start. An example call to this
103     * would be {@code new FastNoise(1337, 0.02f, FastNoise.SIMPLEX_FRACTAL, 4, 0.5f, 2f)}, which makes noise with a
104     * lower frequency, 4 octaves of Simplex noise, and the "inverse" effect on how those octaves work (which makes
105     * the extra added octaves be more significant to the final result and also have a lower frequency, while normally
106     * added octaves have a higher frequency and tend to have a minor effect on the large-scale shape of the noise).
107     * @param seed the int seed for the noise, which should significantly affect the produced noise
108     * @param frequency the multiplier for all dimensions, which is usually fairly small (1.0f/32.0f is the default)
109     * @param noiseType the noiseType, which should be a constant from this class (see {@link #setNoiseType(int)})
110     * @param octaves how many octaves of noise to use when the noiseType is one of the _FRACTAL types
111     * @param lacunarity typically 2.0, or 0.5 to change how extra octaves work (inverse mode) 
112     * @param gain typically 0.5, or 2.0 to change how extra octaves work (inverse mode)
113     */
114    public FastNoise(int seed, float frequency, int noiseType, int octaves, float lacunarity, float gain)
115    {
116        this.seed = seed;
117        this.frequency = frequency;
118        this.noiseType = noiseType;
119        this.octaves = octaves;
120        this.lacunarity = lacunarity;
121        this.gain = gain;
122        calculateFractalBounding();
123    }

1555    private float singleSimplex(int seed, float x, float y, float z) {
1556        float t = (x + y + z) * F3;
1557        int i = fastFloor(x + t);
1558        int j = fastFloor(y + t);
1559        int k = fastFloor(z + t);
1560
1561        t = (i + j + k) * G3;
1562        float x0 = x - (i - t);
1563        float y0 = y - (j - t);
1564        float z0 = z - (k - t);
1565
1566        int i1, j1, k1;
1567        int i2, j2, k2;
1568
1569        if (x0 >= y0) {
1570            if (y0 >= z0) {
1571                i1 = 1;
1572                j1 = 0;
1573                k1 = 0;
1574                i2 = 1;
1575                j2 = 1;
1576                k2 = 0;
1577            } else if (x0 >= z0) {
1578                i1 = 1;
1579                j1 = 0;
1580                k1 = 0;
1581                i2 = 1;
1582                j2 = 0;
1583                k2 = 1;
1584            } else // x0 < z0
1585            {
1586                i1 = 0;
1587                j1 = 0;
1588                k1 = 1;
1589                i2 = 1;
1590                j2 = 0;
1591                k2 = 1;
1592            }
1593        } else // x0 < y0
1594        {
1595            if (y0 < z0) {
1596                i1 = 0;
1597                j1 = 0;
1598                k1 = 1;
1599                i2 = 0;
1600                j2 = 1;
1601                k2 = 1;
1602            } else if (x0 < z0) {
1603                i1 = 0;
1604                j1 = 1;
1605                k1 = 0;
1606                i2 = 0;
1607                j2 = 1;
1608                k2 = 1;
1609            } else // x0 >= z0
1610            {
1611                i1 = 0;
1612                j1 = 1;
1613                k1 = 0;
1614                i2 = 1;
1615                j2 = 1;
1616                k2 = 0;
1617            }
1618        }
1619
1620        float x1 = x0 - i1 + G3;
1621        float y1 = y0 - j1 + G3;
1622        float z1 = z0 - k1 + G3;
1623        float x2 = x0 - i2 + F3;
1624        float y2 = y0 - j2 + F3;
1625        float z2 = z0 - k2 + F3;
1626        float x3 = x0 + G33;
1627        float y3 = y0 + G33;
1628        float z3 = z0 + G33;
1629
1630        float n = 0;
1631
1632        t = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
1633        if (t > 0) {
1634            t *= t;
1635            n += t * t * gradCoord3D(seed, i, j, k, x0, y0, z0);
1636        }
1637
1638        t = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
1639        if (t > 0) {
1640            t *= t;
1641            n += t * t * gradCoord3D(seed, i + i1, j + j1, k + k1, x1, y1, z1);
1642        }
1643
1644        t = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
1645        if (t > 0) {
1646            t *= t;
1647            n += t * t * gradCoord3D(seed, i + i2, j + j2, k + k2, x2, y2, z2);
1648        }
1649

1652            t *= t;

1654        }
